#include <iostream>
#include <string>
using namespace std;

string timeConversion(string s) {
    string hourStr = s.substr(0, 2);
    int hour = stoi(hourStr);

    string period = s.substr(8, 2);

    if (period == "AM") {
        if (hour == 12) hour = 0;
    } else {
        if (hour != 12) hour += 12; 
    }
    string result = (hour < 10 ? "0" : "") + to_string(hour) + s.substr(2, 6);
    return result;
}

int main() {
    string s;
    cin >> s; 
    cout << timeConversion(s) << endl;
    return 0;
}
