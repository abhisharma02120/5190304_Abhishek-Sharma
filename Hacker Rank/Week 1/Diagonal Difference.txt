#include <bits/stdc++.h>
using namespace std;
int diagonalDifference(vector<vector<int>> arr) {
    int n = arr.size();
    int primary = 0, secondary = 0;
    for (int i = 0; i < n; i++) {
        primary += arr[i][i];                
        secondary += arr[i][n - i - 1];      
    }
    return abs(primary - secondary);
}
int main() {
    int n;
    cin >> n;
    vector<vector<int>> arr(n, vector<int>(n));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> arr[i][j];
        }
    }
    cout << diagonalDifference(arr) << endl;
    return 0;
}
