#include <bits/stdc++.h>
using namespace std;

int lonelyinteger(vector<int> a) {
    int result = 0;
    for (int num : a) {
        result ^= num;  
    }
    return result;
}

int main() {
    int n;
    cin >> n;
    vector<int> a(n);
    
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    cout << lonelyinteger(a) << endl;
    return 0;
}
